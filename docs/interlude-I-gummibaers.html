gummi_count_tbl <- gummi_bears_tbl %>%
  rowwise() %>% 
  mutate(count_bear = sum(c_across(darkred:white)),
         count_color = sum(c_across(darkred:white) > 0)) |> 
  ungroup() |> 
  select(year, count_bear, count_color)
